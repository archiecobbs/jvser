008    package org.dellroad.jvser;
009    
010    import static org.dellroad.jvser.RFC2217.COM_PORT_OPTION;
011    import static org.dellroad.jvser.RFC2217.DATASIZE_5;
012    import static org.dellroad.jvser.RFC2217.DATASIZE_6;
013    import static org.dellroad.jvser.RFC2217.DATASIZE_7;
014    import static org.dellroad.jvser.RFC2217.DATASIZE_8;
015    import static org.dellroad.jvser.RFC2217.DATASIZE_REQUEST;
016    import static org.dellroad.jvser.RFC2217.SERVER_OFFSET;
017    import static org.dellroad.jvser.RFC2217.SET_DATASIZE;
018    
019    /**
020     * RFC 2217 {@code SET-DATASIZE} command.
021     *
022     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
023     */
024    public class DataSizeCommand extends ComPortCommand {
025    
026        private int dataSize;
027    
028        /**
029         * Decoding constructor.
030         *
031         * @param bytes encoded option starting with the {@code COM-PORT-OPTION} byte
032         * @throws NullPointerException if {@code bytes} is null
033         * @throws IllegalArgumentException if {@code bytes} has length != 3
034         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
035         * @throws IllegalArgumentException if {@code bytes[1]} is not {@link RFC2217#SET_DATASIZE} (client or server)
036         * @throws IllegalArgumentException if {@code bytes[2]} is not a valid RFC 2217 data size value
037         */
038        public DataSizeCommand(int[] bytes) {
039            super("SET-DATASIZE", SET_DATASIZE, bytes);
040            this.dataSize = bytes[2];
041            switch (this.dataSize) {
042            case DATASIZE_REQUEST:
043            case DATASIZE_5:
044            case DATASIZE_6:
045            case DATASIZE_7:
046            case DATASIZE_8:
047                break;
048            default:
049                throw new IllegalArgumentException("invalid data size value " + this.dataSize);
050            }
051        }
052    
053        /**
054         * Encoding constructor.
055         *
056         * @param dataSize data size value
057         * @param client true for the client-to-server command, false for the server-to-client command
058         * @throws IllegalArgumentException if {@code dataSize} is not a valid RFC 2217 data size value
059         */
060        public DataSizeCommand(boolean client, int dataSize) {
061            this(new int[] {
062                COM_PORT_OPTION,
063                client ? SET_DATASIZE : SET_DATASIZE + SERVER_OFFSET,
064                dataSize
065            });
066        }
067    
068        @Override
069        public String toString() {
070            String desc;
071            switch (this.dataSize) {
072            case DATASIZE_REQUEST:
073                desc = "REQUEST";
074                break;
075            case DATASIZE_5:
076                desc = "5";
077                break;
078            case DATASIZE_6:
079                desc = "6";
080                break;
081            case DATASIZE_7:
082                desc = "7";
083                break;
084            case DATASIZE_8:
085                desc = "8";
086                break;
087            default:
088                desc = "?";
089                break;
090            }
091            return this.getName() + " " + desc;
092        }
093    
094        @Override
095        public void visit(ComPortCommandSwitch sw) {
096            sw.caseDataSize(this);
097        }
098    
099        public int getDataSize() {
100            return this.dataSize;
101        }
102    
103        @Override
104        int getMinPayloadLength() {
105            return 1;
106        }
107    
108        @Override
109        int getMaxPayloadLength() {
110            return 1;
111        }
112    }
