109        /**
110         * Decode an RFC 2217 {@code COM-PORT-OPTION} command.
111         *
112         * @throws IllegalArgumentException if the bytes are not a valid encoded RFC 2217 {@link #COM_PORT_OPTION}
113         */
114        public static ComPortCommand decodeComPortCommand(int[] bytes) {
115            if (bytes.length < 2)
116                throw new IllegalArgumentException("length < 2");
117            if (bytes[0] != COM_PORT_OPTION)
118                throw new IllegalArgumentException("not a COM-PORT-OPTION (option = " + bytes[0] + ")");
119            switch (bytes[1]) {
120            case SIGNATURE:
121            case SIGNATURE + SERVER_OFFSET:
122                return new SignatureCommand(bytes);
123            case SET_BAUDRATE:
124            case SET_BAUDRATE + SERVER_OFFSET:
125                return new BaudRateCommand(bytes);
126            case SET_DATASIZE:
127            case SET_DATASIZE + SERVER_OFFSET:
128                return new DataSizeCommand(bytes);
129            case SET_PARITY:
130            case SET_PARITY + SERVER_OFFSET:
131                return new ParityCommand(bytes);
132            case SET_STOPSIZE:
133            case SET_STOPSIZE + SERVER_OFFSET:
134                return new StopSizeCommand(bytes);
135            case SET_CONTROL:
136            case SET_CONTROL + SERVER_OFFSET:
137                return new ControlCommand(bytes);
138            case NOTIFY_LINESTATE:
139            case NOTIFY_LINESTATE + SERVER_OFFSET:
140                return new NotifyLineStateCommand(bytes);
141            case NOTIFY_MODEMSTATE:
142            case NOTIFY_MODEMSTATE + SERVER_OFFSET:
143                return new NotifyModemStateCommand(bytes);
144            case FLOWCONTROL_SUSPEND:
145            case FLOWCONTROL_SUSPEND + SERVER_OFFSET:
146                return new FlowControlSuspendCommand(bytes);
147            case FLOWCONTROL_RESUME:
148            case FLOWCONTROL_RESUME + SERVER_OFFSET:
149                return new FlowControlResumeCommand(bytes);
150            case SET_LINESTATE_MASK:
151            case SET_LINESTATE_MASK + SERVER_OFFSET:
152                return new LineStateMaskCommand(bytes);
153            case SET_MODEMSTATE_MASK:
154            case SET_MODEMSTATE_MASK + SERVER_OFFSET:
155                return new ModemStateMaskCommand(bytes);
156            case PURGE_DATA:
157            case PURGE_DATA + SERVER_OFFSET:
158                return new PurgeDataCommand(bytes);
159            default:
160                throw new IllegalArgumentException("unrecognized COM-PORT-OPTION command " + bytes[1]);
161            }
162        }
