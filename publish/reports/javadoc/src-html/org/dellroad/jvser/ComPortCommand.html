008    package org.dellroad.jvser;
009    
010    import java.util.Arrays;
011    
012    import static org.dellroad.jvser.RFC2217.COM_PORT_OPTION;
013    import static org.dellroad.jvser.RFC2217.SERVER_OFFSET;
014    
015    /**
016     * Superclass for RFC 2217 commands.
017     *
018     * <p>
019     * Instances of this class (and all subclasses) are immutable.
020     * </p>
021     *
022     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
023     */
024    public abstract class ComPortCommand {
025    
026        final String name;
027        final int[] bytes;
028    
029        /**
030         * Constructor.
031         *
032         * @param name human readable name of this command
033         * @param command required {@code COM-PORT-OPTION} command byte value (must be the client-to-server value)
034         * @param bytes encoded command starting with the {@code COM-PORT-OPTION} byte
035         * @throws NullPointerException if {@code bytes} is null
036         * @throws IllegalArgumentException if {@code bytes} has length that is too short or too long
037         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
038         * @throws IllegalArgumentException if {@code command} is greater than or equal to {@link RFC2217#SERVER_OFFSET}
039         * @throws IllegalArgumentException if {@code bytes[1]} is not {@code command}
040         *  or {@code command} + {@link RFC2217#SERVER_OFFSET}
041         */
042        protected ComPortCommand(String name, int command, int[] bytes) {
043            this.name = name;
044            int minLength = 2 + this.getMinPayloadLength();
045            int maxLength = 2 + this.getMaxPayloadLength();
046            if (bytes.length < minLength || bytes.length > maxLength) {
047                throw new IllegalArgumentException("command " + command + " length = "
048                  + bytes.length + " is not in the range " + minLength + ".." + maxLength);
049            }
050            this.bytes = bytes.clone();                 // maintain immutability
051            if (this.bytes[0] != COM_PORT_OPTION)
052                throw new IllegalArgumentException("not a COM-PORT-OPTION");
053            if (command >= SERVER_OFFSET)
054                throw new IllegalArgumentException("invalid command " + command);
055            if (this.getCommand() != command && this.getCommand() != command + SERVER_OFFSET)
056                throw new IllegalArgumentException("not a " + name + " option");
057        }
058    
059        /**
060         * Determine if this command is client-to-server or server-to-client.
061         *
062         * @return true if this command is sent from the server to the client, false for the opposite
063         */
064        public final boolean isServerCommand() {
065            return this.getCommand() >= SERVER_OFFSET;
066        }
067    
068        /**
069         * Get the encoding of this instance.
070         *
071         * @return encoding starting with {@code COM-PORT-OPTION}
072         */
073        public final int[] getBytes() {
074            return this.bytes.clone();                  // maintain immutability
075        }
076    
077        /**
078         * Get the command byte.
079         *
080         * @return RFC 2217-defined byte value for this command
081         */
082        public final int getCommand() {
083            return this.bytes[1] & 0xff;
084        }
085    
086        /**
087         * Get the human-readable name of this option.
088         */
089        public String getName() {
090            return this.name + (this.isServerCommand() ? "[S]" : "[C]");
091        }
092    
093        /**
094         * Get the human-readable description of this option.
095         */
096        @Override
097        public abstract String toString();
098    
099        /**
100         * Apply visitor pattern.
101         *
102         * @param sw visitor switch handler
103         */
104        public abstract void visit(ComPortCommandSwitch sw);
105    
106        /**
107         * Get the option payload as bytes.
108         */
109        byte[] getPayload() {
110            byte[] buf = new byte[this.bytes.length - 2];
111            for (int i = 2; i < bytes.length; i++)
112                buf[i - 2] = (byte)this.bytes[i];
113            return buf;
114        }
115    
116        /**
117         * Get minimum required length of the payload of this command.
118         */
119        abstract int getMinPayloadLength();
120    
121        /**
122         * Get maximum required length of the payload of this command.
123         */
124        abstract int getMaxPayloadLength();
125    
126        @Override
127        public boolean equals(Object obj) {
128            if (obj == null || obj.getClass() != getClass())
129                return false;
130            ComPortCommand that = (ComPortCommand)obj;
131            return Arrays.equals(this.bytes, that.bytes);
132        }
133    
134        @Override
135        public int hashCode() {
136            int hash = 0;
137            for (int i = 0; i < this.bytes.length; i++)
138                hash = hash * 37 + this.bytes[i];
139            return hash;
140        }
141    }
