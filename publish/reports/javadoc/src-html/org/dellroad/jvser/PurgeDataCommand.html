008    package org.dellroad.jvser;
009    
010    import static org.dellroad.jvser.RFC2217.*;
011    
012    /**
013     * RFC 2217 {@code PURGE-DATA} command.
014     *
015     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
016     */
017    public class PurgeDataCommand extends ComPortCommand {
018    
019        private int purgeData;
020    
021        /**
022         * Decoding constructor.
023         *
024         * @param bytes encoded option starting with the {@code COM-PORT-OPTION} byte
025         * @throws NullPointerException if {@code bytes} is null
026         * @throws IllegalArgumentException if {@code bytes} has length != 3
027         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
028         * @throws IllegalArgumentException if {@code bytes[1]} is not {@link RFC2217#PURGE_DATA} (client or server)
029         * @throws IllegalArgumentException if {@code bytes[2]} is not a valid RFC 2217 purge data value
030         */
031        public PurgeDataCommand(int[] bytes) {
032            super("PURGE-DATA", PURGE_DATA, bytes);
033            this.purgeData = bytes[2];
034            switch (this.purgeData) {
035            case PURGE_DATA_RECEIVE_DATA_BUFFER:
036            case PURGE_DATA_TRANSMIT_DATA_BUFFER:
037            case PURGE_DATA_BOTH_DATA_BUFFERS:
038                break;
039            default:
040                throw new IllegalArgumentException("invalid purge data value " + this.purgeData);
041            }
042        }
043    
044        /**
045         * Encoding constructor.
046         *
047         * @param purgeData purge data value
048         * @param client true for the client-to-server command, false for the server-to-client command
049         * @throws IllegalArgumentException if {@code purgeData} is not a valid RFC 2217 purge data value
050         */
051        public PurgeDataCommand(boolean client, int purgeData) {
052            this(new int[] {
053                COM_PORT_OPTION,
054                client ? PURGE_DATA : PURGE_DATA + SERVER_OFFSET,
055                purgeData
056            });
057        }
058    
059        @Override
060        public String toString() {
061            String desc;
062            switch (this.purgeData) {
063            case PURGE_DATA_RECEIVE_DATA_BUFFER:
064                desc = "RECEIVE-DATA-BUFFER";
065                break;
066            case PURGE_DATA_TRANSMIT_DATA_BUFFER:
067                desc = "TRANSMIT-DATA-BUFFER";
068                break;
069            case PURGE_DATA_BOTH_DATA_BUFFERS:
070                desc = "BOTH-DATA-BUFFERS";
071                break;
072            default:
073                desc = "?";
074                break;
075            }
076            return this.getName() + " " + desc;
077        }
078    
079        @Override
080        public void visit(ComPortCommandSwitch sw) {
081            sw.casePurgeData(this);
082        }
083    
084        public boolean isPurgeReceiveDataBuffer() {
085            return (this.purgeData & PURGE_DATA_RECEIVE_DATA_BUFFER) != 0;
086        }
087    
088        public boolean isPurgeTransmitDataBuffer() {
089            return (this.purgeData & PURGE_DATA_TRANSMIT_DATA_BUFFER) != 0;
090        }
091    
092        @Override
093        int getMinPayloadLength() {
094            return 1;
095        }
096    
097        @Override
098        int getMaxPayloadLength() {
099            return 1;
100        }
101    }
