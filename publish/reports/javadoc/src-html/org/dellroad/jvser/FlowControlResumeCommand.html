008    package org.dellroad.jvser;
009    
010    import static org.dellroad.jvser.RFC2217.*;
011    
012    /**
013     * RFC 2217 {@code FLOWCONTROL-RESUME} command.
014     *
015     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
016     */
017    public class FlowControlResumeCommand extends ComPortCommand {
018    
019        /**
020         * Decoding constructor.
021         *
022         * @param bytes encoded option starting with the {@code COM-PORT-OPTION} byte
023         * @throws NullPointerException if {@code bytes} is null
024         * @throws IllegalArgumentException if {@code bytes} has length != 3
025         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
026         * @throws IllegalArgumentException if {@code bytes[1]} is not {@link RFC2217#FLOWCONTROL_RESUME} (client or server)
027         */
028        public FlowControlResumeCommand(int[] bytes) {
029            super("FLOWCONTROL-RESUME", FLOWCONTROL_RESUME, bytes);
030        }
031    
032        /**
033         * Encoding constructor.
034         *
035         * @param client true for the client-to-server command, false for the server-to-client command
036         */
037        public FlowControlResumeCommand(boolean client) {
038            this(new int[] {
039                COM_PORT_OPTION,
040                client ? FLOWCONTROL_RESUME : FLOWCONTROL_RESUME + SERVER_OFFSET,
041            });
042        }
043    
044        @Override
045        public String toString() {
046            return this.getName();
047        }
048    
049        @Override
050        public void visit(ComPortCommandSwitch sw) {
051            sw.caseFlowControlResume(this);
052        }
053    
054        @Override
055        int getMinPayloadLength() {
056            return 0;
057        }
058    
059        @Override
060        int getMaxPayloadLength() {
061            return 0;
062        }
063    }
