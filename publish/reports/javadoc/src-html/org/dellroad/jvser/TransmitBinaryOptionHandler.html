022            super(TRANSMIT_BINARY_OPTION, initlocal, initremote, acceptlocal, acceptremote);
023        }
024    
025        @Override
026        public int[] answerSubnegotiation(int[] data, int length) {
027            return null;
028        }
029

032            return null;
033        }
034    
035        @Override
036        public int[] startSubnegotiationRemote() {
037            return null;
038        }
039    }
