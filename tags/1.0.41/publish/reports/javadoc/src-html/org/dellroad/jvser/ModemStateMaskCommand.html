008    package org.dellroad.jvser;
009    
010    import static org.dellroad.jvser.RFC2217.COM_PORT_OPTION;
011    import static org.dellroad.jvser.RFC2217.SERVER_OFFSET;
012    import static org.dellroad.jvser.RFC2217.SET_MODEMSTATE_MASK;
013    
014    /**
015     * RFC 2217 {@code SET-MODEMSTATE-MASK} command.
016     *
017     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
018     */
019    public class ModemStateMaskCommand extends ComPortCommand {
020    
021        private int modemStateMask;
022    
023        /**
024         * Decoding constructor.
025         *
026         * @param bytes encoded option starting with the {@code COM-PORT-OPTION} byte
027         * @throws NullPointerException if {@code bytes} is null
028         * @throws IllegalArgumentException if {@code bytes} has length != 3
029         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
030         * @throws IllegalArgumentException if {@code bytes[1]} is not {@link RFC2217#SET_MODEMSTATE_MASK} (client or server)
031         */
032        public ModemStateMaskCommand(int[] bytes) {
033            super("SET-MODEMSTATE-MASK", SET_MODEMSTATE_MASK, bytes);
034            this.modemStateMask = bytes[2];
035        }
036    
037        /**
038         * Encoding constructor.
039         *
040         * @param modemStateMask modem state mask value
041         * @param client true for the client-to-server command, false for the server-to-client command
042         */
043        public ModemStateMaskCommand(boolean client, int modemStateMask) {
044            this(new int[] {
045                COM_PORT_OPTION,
046                client ? SET_MODEMSTATE_MASK : SET_MODEMSTATE_MASK + SERVER_OFFSET,
047                modemStateMask
048            });
049        }
050    
051        @Override
052        public String toString() {
053            return this.getName() + " " + Util.decodeBits(this.modemStateMask, Util.MODEM_STATE_BITS);
054        }
055    
056        @Override
057        public void visit(ComPortCommandSwitch sw) {
058            sw.caseModemStateMask(this);
059        }
060    
061        public int getModemStateMask() {
062            return this.modemStateMask;
063        }
064    
065        @Override
066        int getMinPayloadLength() {
067            return 1;
068        }
069    
070        @Override
071        int getMaxPayloadLength() {
072            return 1;
073        }
074    }
