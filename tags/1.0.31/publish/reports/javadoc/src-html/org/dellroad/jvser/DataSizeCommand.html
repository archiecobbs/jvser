008    package org.dellroad.jvser;
009    
010    import static org.dellroad.jvser.RFC2217.*;
011    
012    /**
013     * RFC 2217 {@code SET-DATASIZE} command.
014     *
015     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
016     */
017    public class DataSizeCommand extends ComPortCommand {
018    
019        private int dataSize;
020    
021        /**
022         * Decoding constructor.
023         *
024         * @param bytes encoded option starting with the {@code COM-PORT-OPTION} byte
025         * @throws NullPointerException if {@code bytes} is null
026         * @throws IllegalArgumentException if {@code bytes} has length != 3
027         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
028         * @throws IllegalArgumentException if {@code bytes[1]} is not {@link RFC2217#SET_DATASIZE} (client or server)
029         * @throws IllegalArgumentException if {@code bytes[2]} is not a valid RFC 2217 data size value
030         */
031        public DataSizeCommand(int[] bytes) {
032            super("SET-DATASIZE", SET_DATASIZE, bytes);
033            this.dataSize = bytes[2];
034            switch (this.dataSize) {
035            case DATASIZE_REQUEST:
036            case DATASIZE_5:
037            case DATASIZE_6:
038            case DATASIZE_7:
039            case DATASIZE_8:
040                break;
041            default:
042                throw new IllegalArgumentException("invalid data size value " + this.dataSize);
043            }
044        }
045    
046        /**
047         * Encoding constructor.
048         *
049         * @param dataSize data size value
050         * @param client true for the client-to-server command, false for the server-to-client command
051         * @throws IllegalArgumentException if {@code dataSize} is not a valid RFC 2217 data size value
052         */
053        public DataSizeCommand(boolean client, int dataSize) {
054            this(new int[] {
055                COM_PORT_OPTION,
056                client ? SET_DATASIZE : SET_DATASIZE + SERVER_OFFSET,
057                dataSize
058            });
059        }
060    
061        @Override
062        public String toString() {
063            String desc;
064            switch (this.dataSize) {
065            case DATASIZE_REQUEST:
066                desc = "REQUEST";
067                break;
068            case DATASIZE_5:
069                desc = "5";
070                break;
071            case DATASIZE_6:
072                desc = "6";
073                break;
074            case DATASIZE_7:
075                desc = "7";
076                break;
077            case DATASIZE_8:
078                desc = "8";
079                break;
080            default:
081                desc = "?";
082                break;
083            }
084            return this.getName() + " " + desc;
085        }
086    
087        @Override
088        public void visit(ComPortCommandSwitch sw) {
089            sw.caseDataSize(this);
090        }
091    
092        public int getDataSize() {
093            return this.dataSize;
094        }
095    
096        @Override
097        int getMinPayloadLength() {
098            return 1;
099        }
100    
101        @Override
102        int getMaxPayloadLength() {
103            return 1;
104        }
105    }
