029    /**
030     * Implements the client side of the <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
031     * serial-over-Telnet protocol as as {@link SerialPort}.
032     *
033     * <p>
034     * This class extends the {@link SerialPort} class and functions in the same way, however, there are
035     * a couple of differences to be aware of:
036     * <ul>
037     * <li>
038     * To "open" a serial port, create an instance of this class, configure it as required,
039     * and then get the {@link TelnetClient} via {@link #getTelnetClient} and invoke
040     * {@link TelnetClient#connect(java.net.InetAddress, int) TelnetClient.connect()} (or one of its variants).
041     * This will create the telnet connection to the access server.
042     * </li>
043     *
044     * <li>
045     * Once connected, if the underlying telnet connection is broken, an {@link IOException} will be
046     * thrown when attempting to access the serial port input or output streams. In addition, a
047     * {@link SerialPortEvent#DATA_AVAILABLE DATA_AVAILABLE} event will be immediately generated.
048     * </li>
049     * </ul>
050     * </p>
051     *
052     * <p>
053     * The following optional functionality is not implemented and/or inappropriate for a networked connection:
054     * <ul>
055     * <li>Receive threshold</li>
056     * <li>Receive timeout</li>
057     * <li>Receive framing byte</li>
058     * <li>Input buffer size</li>
059     * <li>Output buffer size</li>
060     * </ul>
061     * </p>
062     *
063     * <p>
064     * In addition, access servers typically don't support {@link #notifyOnOutputEmpty}.
065     * </p>
066     *
067     * <p>
068     * Finally, {@link #sendBreak} is supported but the {@code millis} argument is ignored, as timing cannot be
069     * assured over a TCP connection. Access servers typically enforce a fixed break time.
070     * </p>
071     *
072     * @see SerialPort
073     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
074     */
075    public class TelnetSerialPort extends SerialPort {

204         * Construct and configure the {@link TelnetClient} to be used for this instance.
205         */
206        protected TelnetClient createTelnetClient() {
207            TelnetClient tc = new TelnetClient(DEFAULT_TERMINAL_TYPE);
208            tc.setReaderThread(true);                                   // allows immediate option negotiation
209            try {
210                tc.addOptionHandler(new TerminalTypeOptionHandler(DEFAULT_TERMINAL_TYPE, false, false, true, false));
211                tc.addOptionHandler(new EchoOptionHandler(false, false, false, false));
212                tc.addOptionHandler(new SuppressGAOptionHandler(true, true, true, true));
213                tc.addOptionHandler(new TransmitBinaryOptionHandler(true, true, true, true));
214                tc.addOptionHandler(new ComPortOptionHandler(this));
215            } catch (InvalidTelnetOptionException e) {
216                throw new RuntimeException("unexpected exception", e);
217            }
218            return tc;
219        }

238            if (this.state == State.CLOSED)

308            synchronized (this) {

310                if (this.state != State.ESTABLISHED)

317        @Override
318        public synchronized void setFlowControlMode(int flowControl) throws UnsupportedCommOperationException {
319            this.state.checkNotClosed();
320    
321            // Validate bit combination
322            if ((flowControl & (FLOWCONTROL_RTSCTS_OUT | FLOWCONTROL_XONXOFF_OUT)) == (FLOWCONTROL_RTSCTS_OUT | FLOWCONTROL_XONXOFF_OUT)
323             || (flowControl & (FLOWCONTROL_RTSCTS_IN | FLOWCONTROL_XONXOFF_IN)) == (FLOWCONTROL_RTSCTS_IN | FLOWCONTROL_XONXOFF_IN))
324                throw new UnsupportedCommOperationException("invalid flow control value " + flowControl);
325    
326            // Convert to RFC 2217 values
327            int previousFlowControlOutbound = this.flowControlOutbound;
328            int previousFlowControlInbound = this.flowControlInbound;
329            this.flowControlOutbound = (flowControl & FLOWCONTROL_RTSCTS_OUT) != 0 ? CONTROL_OUTBOUND_FLOW_HARDWARE :
330              (flowControl & FLOWCONTROL_XONXOFF_OUT) != 0 ? CONTROL_OUTBOUND_FLOW_XON_XOFF : CONTROL_OUTBOUND_FLOW_NONE;
331            this.flowControlInbound = (flowControl & FLOWCONTROL_RTSCTS_IN) != 0 ? CONTROL_INBOUND_FLOW_HARDWARE :
332              (flowControl & FLOWCONTROL_XONXOFF_IN) != 0 ? CONTROL_INBOUND_FLOW_XON_XOFF : CONTROL_INBOUND_FLOW_NONE;
333    
334            // Update server (outbound first per RFC 2217)
335            if (this.flowControlOutbound != previousFlowControlOutbound && this.state.isEstablished())
336                this.sendSubnegotiation(new ControlCommand(true, this.flowControlOutbound));
337            if (this.flowControlInbound != previousFlowControlInbound && this.state.isEstablished())
338                this.sendSubnegotiation(new ControlCommand(true, this.flowControlInbound));
339        }

341        @Override
342        public synchronized int getFlowControlMode() {
343            this.state.checkNotClosed();
344            int value = FLOWCONTROL_NONE;
345            switch (this.flowControlOutbound) {
346            case CONTROL_OUTBOUND_FLOW_HARDWARE:
347                value |= FLOWCONTROL_RTSCTS_OUT;
348                break;
349            case CONTROL_OUTBOUND_FLOW_XON_XOFF:
350                value |= FLOWCONTROL_XONXOFF_OUT;
351                break;
352            default:
353                break;
354            }
355            switch (this.flowControlInbound) {
356            case CONTROL_INBOUND_FLOW_HARDWARE:
357                value |= FLOWCONTROL_RTSCTS_IN;
358                break;
359            case CONTROL_INBOUND_FLOW_XON_XOFF:
360                value |= FLOWCONTROL_XONXOFF_IN;
361                break;
362            default:
363                break;
364            }
365            return value;
366        }

368        @Override
369        public synchronized void setSerialPortParams(int baudRate, int dataBits, int stopBits, int parity)
370          throws UnsupportedCommOperationException {
371            this.state.checkNotClosed();
372    
373            // Validate parameters and convert to RFC 2217 values
374            if (baudRate <= 0)
375                throw new UnsupportedCommOperationException("invalid baud rate " + baudRate);
376            switch (dataBits) {
377            case DATABITS_5:
378                dataBits = DATASIZE_5;
379                break;
380            case DATABITS_6:
381                dataBits = DATASIZE_6;
382                break;
383            case DATABITS_7:
384                dataBits = DATASIZE_7;
385                break;
386            case DATABITS_8:
387                dataBits = DATASIZE_8;
388                break;
389            default:
390                throw new UnsupportedCommOperationException("invalid data bits " + dataBits);
391            }
392            switch (stopBits) {
393            case STOPBITS_1:
394                stopBits = STOPSIZE_1;
395                break;
396            case STOPBITS_2:
397                stopBits = STOPSIZE_2;
398                break;
399            case STOPBITS_1_5:
400                stopBits = STOPSIZE_1_5;
401                break;
402            default:
403                throw new UnsupportedCommOperationException("invalid stop bits " + stopBits);
404            }
405            switch (parity) {
406            case SerialPort.PARITY_NONE:
407                parity = RFC2217.PARITY_NONE;
408                break;
409            case SerialPort.PARITY_ODD:
410                parity = RFC2217.PARITY_ODD;
411                break;
412            case SerialPort.PARITY_EVEN:
413                parity = RFC2217.PARITY_EVEN;
414                break;
415            case SerialPort.PARITY_MARK:
416                parity = RFC2217.PARITY_MARK;
417                break;
418            case SerialPort.PARITY_SPACE:
419                parity = RFC2217.PARITY_SPACE;
420                break;
421            default:
422                throw new UnsupportedCommOperationException("invalid parity " + parity);
423            }
424    
425            // Update my state
426            boolean changed = false;
427            if (this.baudRate != baudRate) {
428                this.baudRate = baudRate;
429                changed = true;
430            }
431            if (this.dataSize != dataBits) {
432                this.dataSize = dataBits;
433                changed = true;
434            }
435            if (this.stopSize != stopBits) {
436                this.stopSize = stopBits;
437                changed = true;
438            }
439            if (this.parity != parity) {
440                this.parity = parity;
441                changed = true;
442            }
443    
444            // Update access server if there was a change
445            if (changed && this.state.isEstablished())
446                this.sendSerialPortGeometry();
447        }

505        // This is invoked by the ComPortOptionHandler once the server has agreed to accept COM-PORT-OPTION subnegotiation commands
506    
507        synchronized void startSubnegotiation() {
508    
509            // Log
510            log.debug(this.name + ": server accepted COM-PORT-OPTION, sending serial configuration to peer");
511    
512            // Update state
513            this.state.checkNotClosed();
514            this.state = State.ESTABLISHED;
515    
516            // Request signature from peer
517            this.sendSubnegotiation(new SignatureCommand(true));
518    
519            // Send signature if desired
520            if (this.signature != null && this.signature.length() > 0)
521                this.sendSubnegotiation(new SignatureCommand(true, this.signature));
522    
523            // Send all configuration information
524            this.sendSerialPortGeometry();
525            this.sendSubnegotiation(new LineStateMaskCommand(true, this.lineStateMask));
526            this.sendSubnegotiation(new ModemStateMaskCommand(true, this.modemStateMask));
527            this.sendSubnegotiation(new ControlCommand(true, this.flowControlInbound));
528            this.sendSubnegotiation(new ControlCommand(true, this.flowControlOutbound));
529            this.sendSubnegotiation(new ControlCommand(true, this.dtr ? CONTROL_DTR_ON : CONTROL_DTR_OFF));
530            this.sendSubnegotiation(new ControlCommand(true, this.rts ? CONTROL_RTS_ON : CONTROL_RTS_OFF));
531        }
532    
533        // Method to send serial port "geometry" in the order recommended by RFC 2217 (section 2)
534    
535        private void sendSerialPortGeometry() {
536            this.sendSubnegotiation(new BaudRateCommand(true, this.baudRate));
537            this.sendSubnegotiation(new DataSizeCommand(true, this.dataSize));
538            this.sendSubnegotiation(new ParityCommand(true, this.parity));
539            this.sendSubnegotiation(new StopSizeCommand(true, this.stopSize));
540        }

542        // This is invoked by the ComPortOptionHandler when we receive a command from the server
543    
544        void handleCommand(ComPortCommand command) {
545    
546            // Incoming commands should be server versions
547            if (!command.isServerCommand()) {
548                log.warn(TelnetSerialPort.this.name + ": rec'd " + command + " (ignoring unexpected client command)");
549                return;
550            }
551    
552            // Handle command
553            command.visit(new AbstractComPortCommandSwitch() {
554    
555                @Override
556                public void caseBaudRate(BaudRateCommand command) {
557                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
558                    synchronized (TelnetSerialPort.this) {
559                        TelnetSerialPort.this.baudRate = command.getBaudRate();
560                    }
561                }
562    
563                @Override
564                public void caseDataSize(DataSizeCommand command) {
565                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
566                    synchronized (TelnetSerialPort.this) {
567                        TelnetSerialPort.this.dataSize = command.getDataSize();
568                    }
569                }
570    
571                @Override
572                public void caseParity(ParityCommand command) {
573                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
574                    synchronized (TelnetSerialPort.this) {
575                        TelnetSerialPort.this.parity = command.getParity();
576                    }
577                }
578    
579                @Override
580                public void caseStopSize(StopSizeCommand command) {
581                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
582                    synchronized (TelnetSerialPort.this) {
583                        TelnetSerialPort.this.stopSize = command.getStopSize();
584                    }
585                }
586    
587                @Override
588                public void caseControl(ControlCommand command) {
589                    synchronized (TelnetSerialPort.this) {
590                        switch (command.getControl()) {
591                        case CONTROL_OUTBOUND_FLOW_NONE:
592                        case CONTROL_OUTBOUND_FLOW_XON_XOFF:
593                        case CONTROL_OUTBOUND_FLOW_HARDWARE:
594                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
595                            TelnetSerialPort.this.flowControlOutbound = command.getControl();
596                            break;
597                        case CONTROL_INBOUND_FLOW_NONE:
598                        case CONTROL_INBOUND_FLOW_XON_XOFF:
599                        case CONTROL_INBOUND_FLOW_HARDWARE:
600                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
601                            TelnetSerialPort.this.flowControlInbound = command.getControl();
602                            break;
603                        case CONTROL_DTR_ON:
604                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
605                            TelnetSerialPort.this.dtr = true;
606                            break;
607                        case CONTROL_DTR_OFF:
608                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
609                            TelnetSerialPort.this.dtr = false;
610                            break;
611                        case CONTROL_RTS_ON:
612                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
613                            TelnetSerialPort.this.rts = true;
614                            break;
615                        case CONTROL_RTS_OFF:
616                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
617                            TelnetSerialPort.this.rts = false;
618                            break;
619                        default:
620                            log.debug(TelnetSerialPort.this.name + ": rec'd " + command + " (ignoring)");
621                            break;
622                        }
623                    }
624                }
625    
626                @Override
627                public void caseNotifyLineState(NotifyLineStateCommand command) {
628                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
629                    int lineState = command.getLineState();
630                    int notify;
631                    synchronized (TelnetSerialPort.this) {
632                        notify = TelnetSerialPort.this.lineStateNotify;
633                        TelnetSerialPort.this.lineStateLast = lineState;
634                    }
635                    notify &= lineState;                                    // notify only if bit is equal to 1
636                    if ((notify & LINESTATE_TRANSFER_SHIFT_REGISTER_EMPTY) != 0)
637                        TelnetSerialPort.this.sendEvent(SerialPortEvent.OUTPUT_BUFFER_EMPTY);
638                    if ((notify & LINESTATE_BREAK_DETECT) != 0)
639                        TelnetSerialPort.this.sendEvent(SerialPortEvent.BI);
640                    if ((notify & LINESTATE_FRAMING_ERROR) != 0)
641                        TelnetSerialPort.this.sendEvent(SerialPortEvent.FE);
642                    if ((notify & LINESTATE_PARITY_ERROR) != 0)
643                        TelnetSerialPort.this.sendEvent(SerialPortEvent.PE);
644                    if ((notify & LINESTATE_OVERRUN_ERROR) != 0)
645                        TelnetSerialPort.this.sendEvent(SerialPortEvent.OE);
646                }
647    
648                @Override
649                public void caseNotifyModemState(NotifyModemStateCommand command) {
650                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command);
651                    int modemState = command.getModemState();
652                    int notify;
653                    synchronized (TelnetSerialPort.this) {
654                        notify = TelnetSerialPort.this.modemStateNotify;
655                        TelnetSerialPort.this.modemStateLast = modemState;
656                    }
657                    notify &= modemState ^ modemStateLast;                  // notify only if bit has changed
658                    if ((notify & MODEMSTATE_CARRIER_DETECT) != 0)
659                        TelnetSerialPort.this.sendEvent(SerialPortEvent.CD, (modemState & MODEMSTATE_CARRIER_DETECT) != 0);
660                    if ((notify & MODEMSTATE_RING_INDICATOR) != 0)
661                        TelnetSerialPort.this.sendEvent(SerialPortEvent.RI, (modemState & MODEMSTATE_RING_INDICATOR) != 0);
662                    if ((notify & MODEMSTATE_DSR) != 0)
663                        TelnetSerialPort.this.sendEvent(SerialPortEvent.DSR, (modemState & MODEMSTATE_DSR) != 0);
664                    if ((notify & MODEMSTATE_CTS) != 0)
665                        TelnetSerialPort.this.sendEvent(SerialPortEvent.CTS, (modemState & MODEMSTATE_CTS) != 0);
666                }
667    
668                @Override
669                protected void caseDefault(ComPortCommand command) {
670                    log.debug(TelnetSerialPort.this.name + ": rec'd " + command + " (ignoring)");
671                }
672            });
673        }

761        // Methods for sending event notifications
762    
763        private void sendEvent(int type) {
764            sendEvent(type, true);
765        }
766    
767        private void sendEvent(int type, boolean newValue) {
768            SerialPortEventListener currentListener;
769            synchronized (this) {
770                currentListener = this.listener;
771            }
772            if (currentListener == null)
773                return;
774            SerialPortEvent event = new SerialPortEvent(this, type, !newValue, newValue);
775            try {
776                currentListener.serialEvent(event);
777            } catch (Exception e) {
778                log.warn(this.name + ": exception from listener " + listener, e);
779            }
780        }

785        private void sendSubnegotiation(ComPortCommand command) {

795        // Update line state notifications; return true if we need to send new mask to access server
796        private synchronized boolean updateLineStateMask(int bit, boolean value) {
797            int previous = this.lineStateMask;
798            if (value) {
799                this.lineStateNotify |= bit;
800                this.lineStateMask |= bit;
801            } else {
802                this.lineStateNotify &= ~bit;
803                this.lineStateMask &= ~bit;
804            }
805            this.lineStateMask &= ~LINESTATE_NEVER_MONITOR;
806            return this.lineStateMask != previous;
807        }
808    
809        // Update modem state notifications; return true if we need to send new mask to access server
810        private synchronized boolean updateModemStateMask(int bit, boolean value) {
811            int previous = this.modemStateMask;
812            if (value) {
813                this.modemStateNotify |= bit;
814                this.modemStateMask |= bit;
815            } else {
816                this.modemStateNotify &= ~bit;
817                this.modemStateMask &= ~bit;
818            }
819            this.modemStateMask |= MODEMSTATE_ALWAYS_MONITOR;
820            return this.modemStateMask != previous;
821        }
