008    package org.dellroad.jvser;
009    
010    
011    import org.apache.log4j.Logger;
012    import org.dellroad.jvser.telnet.TelnetOptionHandler;
013    
014    /**
015     * RFC 2217 telnet COM-PORT-OPTION.
016     *
017     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
018     */
019    public class ComPortOptionHandler extends TelnetOptionHandler {
020    
021        private final Logger log = Logger.getLogger(getClass());
022        private final TelnetSerialPort port;
023    
024        protected ComPortOptionHandler(TelnetSerialPort telnetSerialPort) {
025            super(RFC2217.COM_PORT_OPTION, true, false, true, false);
026            if (telnetSerialPort == null)
027                throw new IllegalArgumentException("null telnetSerialPort");
028            this.port = telnetSerialPort;
029        }
030    
031        @Override
032        public int[] answerSubnegotiation(int[] data, int length) {
033    
034            // Copy data into buffer of the correct size
035            if (data.length != length) {
036                int[] data2 = new int[length];
037                System.arraycopy(data, 0, data2, 0, length);
038                data = data2;
039            }
040    
041            // Decode option
042            ComPortCommand command;
043            try {
044                command = RFC2217.decodeComPortCommand(data);
045            } catch (IllegalArgumentException e) {
046                log.error(this.port.getName() + ": rec'd invalid COM-PORT-OPTION command: " + e.getMessage());
047                return null;
048            }
049    
050            // Notify port
051            this.port.handleCommand(command);
052            return null;
053        }
054    
055        @Override
056        public int[] startSubnegotiationLocal() {
057            this.port.startSubnegotiation();
058            return null;
059        }
060    
061        @Override
062        public int[] startSubnegotiationRemote() {
063            return null;
064        }
065    }
