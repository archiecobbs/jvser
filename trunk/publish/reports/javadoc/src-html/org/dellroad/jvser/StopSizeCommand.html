008    package org.dellroad.jvser;
009    
010    import static org.dellroad.jvser.RFC2217.COM_PORT_OPTION;
011    import static org.dellroad.jvser.RFC2217.SERVER_OFFSET;
012    import static org.dellroad.jvser.RFC2217.SET_STOPSIZE;
013    import static org.dellroad.jvser.RFC2217.STOPSIZE_1;
014    import static org.dellroad.jvser.RFC2217.STOPSIZE_1_5;
015    import static org.dellroad.jvser.RFC2217.STOPSIZE_2;
016    import static org.dellroad.jvser.RFC2217.STOPSIZE_REQUEST;
017    
018    /**
019     * RFC 2217 {@code SET-STOPSIZE} command.
020     *
021     * @see <a href="http://tools.ietf.org/html/rfc2217">RFC 2217</a>
022     */
023    public class StopSizeCommand extends ComPortCommand {
024    
025        private int stopSize;
026    
027        /**
028         * Decoding constructor.
029         *
030         * @param bytes encoded option starting with the {@code COM-PORT-OPTION} byte
031         * @throws NullPointerException if {@code bytes} is null
032         * @throws IllegalArgumentException if {@code bytes} has length != 3
033         * @throws IllegalArgumentException if {@code bytes[0]} is not {@link RFC2217#COM_PORT_OPTION}
034         * @throws IllegalArgumentException if {@code bytes[1]} is not {@link RFC2217#SET_STOPSIZE} (client or server)
035         * @throws IllegalArgumentException if {@code bytes[2]} is not a valid RFC 2217 stop size value
036         */
037        public StopSizeCommand(int[] bytes) {
038            super("SET-STOPSIZE", SET_STOPSIZE, bytes);
039            this.stopSize = bytes[2];
040            switch (this.stopSize) {
041            case STOPSIZE_REQUEST:
042            case STOPSIZE_1:
043            case STOPSIZE_2:
044            case STOPSIZE_1_5:
045                break;
046            default:
047                throw new IllegalArgumentException("invalid stop size value " + this.stopSize);
048            }
049        }
050    
051        /**
052         * Encoding constructor.
053         *
054         * @param stopSize stop size value
055         * @param client true for the client-to-server command, false for the server-to-client command
056         * @throws IllegalArgumentException if {@code stopSize} is not a valid RFC 2217 stop size value
057         */
058        public StopSizeCommand(boolean client, int stopSize) {
059            this(new int[] {
060                COM_PORT_OPTION,
061                client ? SET_STOPSIZE : SET_STOPSIZE + SERVER_OFFSET,
062                stopSize
063            });
064        }
065    
066        @Override
067        public String toString() {
068            String desc;
069            switch (this.stopSize) {
070            case STOPSIZE_REQUEST:
071                desc = "REQUEST";
072                break;
073            case STOPSIZE_1:
074                desc = "1";
075                break;
076            case STOPSIZE_2:
077                desc = "2";
078                break;
079            case STOPSIZE_1_5:
080                desc = "1.5";
081                break;
082            default:
083                desc = "?";
084                break;
085            }
086            return this.getName() + " " + desc;
087        }
088    
089        @Override
090        public void visit(ComPortCommandSwitch sw) {
091            sw.caseStopSize(this);
092        }
093    
094        public int getStopSize() {
095            return this.stopSize;
096        }
097    
098        @Override
099        int getMinPayloadLength() {
100            return 1;
101        }
102    
103        @Override
104        int getMaxPayloadLength() {
105            return 1;
106        }
107    }
